#Data cleaning
#Lowercase column names for better data manipulation
inventory.columns = inventory.columns.str.lower()
products.columns = products.columns.str.lower()
sales.columns = sales.columns.str.lower()
stores.columns = stores.columns.str.lower()

#Eliminate mexican pesos sign($)
products[["product_cost", "product_price"]] = products[["product_cost", "product_price"]].apply(lambda x: x.str[1:])
#Change data types of product cost and price to float
products[["product_cost", "product_price"]] = products[["product_cost", "product_price"]].apply(pd.to_numeric)

#Analysis of the data based on main questions
#Connects to the sales database
conn = sqlite3.connect('sales_products_stores.db')

#Gets the top 5 stores, stores it and prints the results
top_stores_by_sales_query = '''
SELECT store_name, SUM(Units * Product_Price) AS Revenue
FROM sales_complete
GROUP BY store_name
ORDER BY Revenue DESC
LIMIT 5;
'''

top_stores_by_sales = pd.read_sql(top_stores_by_sales_query, conn)
print('Top 5 stores by sales')
print(top_stores_by_sales)

#Gets the top 5 Bes-selling products, stores it and prints the results
bestselling_products_query = '''
SELECT Product_Category, SUM(Units) AS Total_Units
FROM sales_complete
GROUP BY Product_Category
ORDER BY Total_Units DESC;
'''

bestselling_products = pd.read_sql(bestselling_products_query, conn)
print('Top 5 Best-selling products')
print(bestselling_products)

#Gets the total sales by month, stores it and prints the results
sales_by_month_query = '''
SELECT 
    strftime('%Y-%m', date) AS month,
    SUM(Units * CAST(Product_Price AS FLOAT)) AS Revenue
FROM sales_complete
GROUP BY month
ORDER BY month
'''

sales_by_month= pd.read_sql(sales_by_month_query,conn)
print('Sales by month')
print(sales_by_month)

conn.close()
